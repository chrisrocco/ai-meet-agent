  <files>src/cli/commands/list-devices.ts, src/cli/commands/test-audio.ts</files>
  <action>
    **Create `src/cli/commands/list-devices.ts`:**

    ```typescript
    import type { Command } from 'commander';
    import { loadConfig } from '../../config/loader.js';
    import { detectPlatform } from '../../platform/detect.js';
    import { AgentError } from '../../errors/index.js';

    export function registerListDevicesCommand(program: Command): void {
      program
        .command('list-devices')
        .description('List available audio/video devices')
        .option('-c, --config <path>', 'path to config.json')
        .action(async (options: { config?: string }) => {
          try {
            await listDevices(options);
          } catch (err) {
            if (err instanceof AgentError) {
              console.error(`Error: ${err.message}`);
              console.error(`Hint: ${err.hint}`);
              process.exit(err.exitCode);
            }
            console.error(`Unexpected error: ${(err as Error).message}`);
            process.exit(1);
          }
        });
    }
    ```

    The `listDevices` function:
    - Call `detectPlatform()` and `loadConfig(options.config)`
    - **Native Linux path:**
      - Run `pactl list sources short` (capture sources) and `pactl list sinks short` (output sinks) via child_process.execSync
      - Run `v4l2-ctl --list-devices 2>/dev/null` for video devices (wrap in try/catch — v4l2-ctl may not be installed)
      - Print output grouped under headings: "Audio Input Devices:", "Audio Output Devices:", "Video Devices:"
    - **WSL2 path (config echo):**
      - Print "WSL2 detected — showing configured Windows devices:"
      - Print from config: `wsl2.captureDevice` (audio input), `wsl2.outputDevice` (audio output)
      - Print video: "OBS Virtual Camera (configure on Windows)"
      - Print a note: "To change devices, edit config.json wsl2 section"
    - Always print the current config source at the top: "Config: {path or defaults}"

    **Create `src/cli/commands/test-audio.ts`:**

    Migrate logic from `src/cli/test-devices.ts` into a Commander command handler:

    ```typescript
    import type { Command } from 'commander';
    import { loadConfig } from '../../config/loader.js';
    import { DeviceManager } from '../../devices/index.js';
    import { detectPlatform } from '../../platform/detect.js';
    import { AgentError } from '../../errors/index.js';

    export function registerTestAudioCommand(program: Command): void {
      program
        .command('test-audio')
        .description('Verify audio device setup')
        .option('-c, --config <path>', 'path to config.json')
        .action(async (options: { config?: string }) => {
          try {
            await testAudio(options);
          } catch (err) {
            if (err instanceof AgentError) {
              console.error(`Error: ${err.message}`);
              console.error(`Hint: ${err.hint}`);
              process.exit(err.exitCode);
            }
            console.error(`Unexpected error: ${(err as Error).message}`);
            process.exit(2);
          }
        });
    }
    ```

    The `testAudio` function mirrors `src/cli/test-devices.ts` main():
    - `detectPlatform()`, `loadConfig(options.config)`, `new DeviceManager(config, platform)`
    - `manager.startup({ startTestPattern: platform !== 'wsl2' })` — catch errors and exit 1
    - Print device status (camera, sink, mic, test pattern)
    - WSL2: print bridge mode note; Native: run test pattern for 5 seconds
    - `manager.shutdown()`, print success, `process.exit(0)`

    Exit codes: 0=pass, 1=device failure, 2=unexpected error (same as existing test-devices.ts).
